kind: PersistentVolumeClaim
metadata:
  name: pvc-redis-data
  namespace: core-middle
  labels:
    type: redis
spec:
  storageClassName: cluster-share
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 10Gi

apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: redis-server
  namespace: core-middle
  labels:
    app: redis-server
    type: redis
spec:
  serviceName: redis-server
  selector:
    matchLabels:
      app: redis-server
  replicas: 1
  template:
    metadata:
      labels:
        app: redis-server
        type: redis
    spec:
      containers:
        - name: redis-server
          image: redis:7
          command:
            - redis-server
            - /usr/local/etc/redis/redis.conf
            - '--port 6379'
            - "--requirepass $(REDIS_PASSWORD)"
          volumeMounts:
            - name: redis-data
              mountPath: /data
            - name: redis-conf
              mountPath: /usr/local/etc/redis
          ports:
            - containerPort: 6379
              name: redis
          env:
            - name: REDIS_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: secret-redis-root
                  key: password
      volumes:
        - name: redis-data
          persistentVolumeClaim:
            claimName: pvc-redis-data
        - name: redis-conf
          configMap:
            name: redis-conf

apiVersion: v1
kind: Service
metadata:
  labels:
    app: redis-server
    type: redis
  name: redis
  namespace: core-middle
spec:
  ports:
    - port: 6379
  selector:
    app: redis-server

测试

当所有机器部署完成后，使用以下目录查看部署结果：

1

 kubectl get pvc,secrets,configmaps,pods,service -n core-middle -l type=redis

如果一切无误，可使用以下命令测试，正常情况下将会显示 OK 然后退出。

1
2

REDIS_PASSWD=$(kubectl get secrets -n core-middle  secret-redis-root -o jsonpath='{.data.password}' | base64 -d)
kubectl exec -it -n core-middle pods/redis-server-0  -- redis-cli AUTH $REDIS_PASSWD
